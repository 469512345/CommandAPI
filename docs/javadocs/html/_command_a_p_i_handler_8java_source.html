379  private Predicate generatePermissions(String commandName, CommandPermission permission) {
380 
381  // If we've already registered a permission, set it to the "parent" permission.
382  /*
383  * This permission generation setup ONLY works iff: - You register the parent
384  * permission node FIRST. - Example: /mycmd - permission node my.perm /mycmd
385  * <arg> - permission node my.perm.other
386  *
387  * the my.perm.other permission node is revoked for the COMMAND REGISTRATION,
388  * however: - The permission node IS REGISTERED. - The permission node, if used
389  * for an argument (as in this case), will be used for suggestions for said
390  * argument
391  */
392  if (permissionsToFix.containsKey(commandName.toLowerCase())) {
393  if (!permissionsToFix.get(commandName.toLowerCase()).equals(permission)) {
394  permission = permissionsToFix.get(commandName.toLowerCase());
395  }
396  } else {
397  // Add permission to a list to fix conflicts with minecraft:permissions
398  permissionsToFix.put(commandName.toLowerCase(), permission);
399  }
400 
401  final CommandPermission finalPermission = permission;
402 
403  // Register it to the Bukkit permissions registry
404  if (finalPermission.getPermission() != null) {
405  try {
406  Bukkit.getPluginManager().addPermission(new Permission(finalPermission.getPermission()));
407  } catch (IllegalArgumentException e) {
408  }
409  }
410 
411  return (Object clw) -> permissionCheck(nms.getCommandSenderForCLW(clw), finalPermission);
412  }

420  private boolean permissionCheck(CommandSender sender, CommandPermission permission) {
421  if (sender == null) {
422  return true;
423  }
424  if (permission.equals(CommandPermission.NONE)) {
425  return true;
426  } else if (permission.equals(CommandPermission.OP)) {
427  return sender.isOp();
428  } else {
429  return sender.hasPermission(permission.getPermission());
430  }
431  }

486  // Builds our NMS command using the given arguments for this method, then
487  // registers it
488  protected void register(String commandName, CommandPermission permissions, String[] aliases,
489  final LinkedHashMap<String, Argument> args, CustomCommandExecutor executor) throws Exception {
490  if (CommandAPIMain.getConfiguration().hasVerboseOutput()) {
491  // Create a list of argument names
492  StringBuilder builder = new StringBuilder();
493  args.values().forEach(arg -> builder.append("<").append(arg.getClass().getSimpleName()).append("> "));
494  CommandAPIMain.getLog().info("Registering command /" + commandName + " " + builder.toString());
495  }
496 
497  Command command = generateCommand(args, executor);
498 
499  /*
500  * The innermost argument needs to be connected to the executor. Then that
501  * argument needs to be connected to the previous argument etc. Then the first
502  * argument needs to be connected to the command name
503  *
504  * CommandName -> Args1 -> Args2 -> ... -> ArgsN -> Executor
505  */
506 
507  LiteralCommandNode resultantNode;
508  if (args.isEmpty()) {
509  // Link command name to the executor
510  resultantNode = this.dispatcher.register((LiteralArgumentBuilder) getLiteralArgumentBuilder(commandName)
511  .requires(generatePermissions(commandName, permissions)).executes(command));
512 
513  // Register aliases
514  for (String alias : aliases) {
515  if (CommandAPIMain.getConfiguration().hasVerboseOutput()) {
516  CommandAPIMain.getLog().info("Registering alias /" + alias + " -> " + resultantNode.getName());
517  }
518  // TODO: What if we just do
519  // getLiteralArgumentBuilder(alias).redirect(resultantNode)?
520  this.dispatcher.register((LiteralArgumentBuilder) getLiteralArgumentBuilder(alias)
521  .requires(generatePermissions(alias, permissions)).executes(command));
522  }
523  } else {
524 
525  // List of keys for reverse iteration
526  ArrayList<String> keys = new ArrayList<>(args.keySet());
527 
528  // Link the last element to the executor
529  ArgumentBuilder inner;
530  // New scope used here to prevent innerArg accidentally being used below
531  {
532  Argument innerArg = args.get(keys.get(keys.size() - 1));
533  if (innerArg instanceof LiteralArgument) {
534  String str = ((LiteralArgument) innerArg).getLiteral();
535  inner = getLiteralArgumentBuilderArgument(str, innerArg.getArgumentPermission()).executes(command);
536  } else if (innerArg instanceof CustomProvidedArgument) {
537  inner = getRequiredArgumentBuilderWithProvider(keys.get(keys.size() - 1), innerArg.getRawType(),
538  innerArg.getArgumentPermission(),
539  nms.getSuggestionProvider(((CustomProvidedArgument) innerArg).getSuggestionProvider()))
540  .executes(command);
541  } else {
542  inner = getRequiredArgumentBuilderDynamic(keys.get(keys.size() - 1), innerArg,
543  innerArg.getArgumentPermission()).executes(command);
544  }
545  }
546 
547  // Link everything else up, except the first
548  ArgumentBuilder outer = inner;
549  for (int i = keys.size() - 2; i >= 0; i--) {
550  Argument outerArg = args.get(keys.get(i));
551  if (outerArg instanceof LiteralArgument) {
552  String str = ((LiteralArgument) outerArg).getLiteral();
553  outer = getLiteralArgumentBuilderArgument(str, outerArg.getArgumentPermission()).then(outer);
554  } else if (outerArg instanceof CustomProvidedArgument) {
555  outer = getRequiredArgumentBuilderWithProvider(keys.get(i), outerArg.getRawType(),
556  outerArg.getArgumentPermission(),
557  nms.getSuggestionProvider(((CustomProvidedArgument) outerArg).getSuggestionProvider()))
558  .then(outer);
559  } else {
560  outer = getRequiredArgumentBuilderDynamic(keys.get(i), outerArg, outerArg.getArgumentPermission())
561  .then(outer);
562  }
563  }
564 
565  // Link command name to first argument and register
566  resultantNode = this.dispatcher.register((LiteralArgumentBuilder) getLiteralArgumentBuilder(commandName)
567  .requires(generatePermissions(commandName, permissions)).then(outer));
568 
569  // Register aliases
570  for (String alias : aliases) {
571  if (CommandAPIMain.getConfiguration().hasVerboseOutput()) {
572  CommandAPIMain.getLog().info("Registering alias /" + alias + " -> " + resultantNode.getName());
573  }
574  this.dispatcher.register((LiteralArgumentBuilder) getLiteralArgumentBuilder(alias)
575  .requires(generatePermissions(alias, permissions)).then(outer));
576  }
577  }
578 
579  // Produce the commandDispatch.json file for debug purposes
580  if (CommandAPIMain.getConfiguration().willCreateDispatcherFile()) {
581  File file = CommandAPIMain.getDispatcherFile();
582  try {
583  file.createNewFile();
584  } catch (IOException e) {
585  e.printStackTrace(System.out);
586  }
587 
588  nms.createDispatcherFile(nmsServer, file, dispatcher);
589  }
590  }
